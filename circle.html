<p>Enter the radius of the circle in millimeters:</p>

<input type="number" id="radiusInput" min="0" step="0.1">

<button onclick="drawCircle()">Calculate</button>

<canvas id="myCanvas"></canvas>

<p>Area: <span id="areaOutput"></span> square millimeters</p>

<script>

function drawCircle() {

var radius = document.getElementById("radiusInput").value;

var canvas = document.getElementById("myCanvas");

var ctx = canvas.getContext("2d");

ctx.clearRect(0, 0, canvas.width, canvas.height);

ctx.beginPath();

ctx.arc(canvas.width/2, canvas.height/2, radius, 0, 2*Math.PI);

ctx.stroke();

var area = Math.PI * radius * radius;

document.getElementById("areaOutput").innerHTML = area.toFixed(2);

}

</script>
